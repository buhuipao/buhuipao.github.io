head/tail

head -n 10 #取前十项

tail -n 10 #后十项

head -10 #同head -n 10

#head,tail命令里默认输出的开始为正，结尾为负

head -n +10 #输出第1行到10行

head -n -10 #输出第1行到倒数11行，后面留10行
